üîò Basic Buttons: The Foundation of Interaction!

üìã Basic Button Fundamentals

Basic Buttons are the simplest form of user interaction in SwiftUI. They consist of a label (text or view) and an action closure that executes when tapped!

Key Components:

  • Label: What the user sees (text, image, or custom view)
  • Action: What happens when tapped (closure)
  • Accessibility: Built-in support for VoiceOver
  • State: Automatic disabled/enabled states

Complex Multi-State Button

enum ButtonState {
    case idle, loading, success, error
}

struct MultiStateButton: View {
    @State private var buttonState: ButtonState = .idle
    @State private var progress: Double = 0
    
    var body: some View {
        Button {
            performOperation()
        } label: {
            HStack {
                buttonIcon
                Text(buttonText)
            }
            .frame(minWidth: 120)
        }
        .buttonStyle(.borderedProminent)
        .tint(buttonColor)
        .disabled(buttonState == .loading)
        .overlay(
            // Progress overlay for loading state
            if buttonState == .loading {
                ProgressView(value: progress)
                    .progressViewStyle(LinearProgressViewStyle(tint: .white))
                    .scaleEffect(x: 1, y: 0.5)
                    .opacity(0.7)
            }
        )
        .animation(.easeInOut(duration: 0.3), value: buttonState)
    }
    
    @ViewBuilder
    var buttonIcon: some View {
        switch buttonState {
        case .idle:
            Image(systemName: "arrow.down.circle")
        case .loading:
            ProgressView()
                .scaleEffect(0.8)
                .progressViewStyle(CircularProgressViewStyle(tint: .white))
        case .success:
            Image(systemName: "checkmark.circle")
        case .error:
            Image(systemName: "exclamationmark.triangle")
        }
    }
    
    var buttonText: String {
        switch buttonState {
        case .idle: return "Download"
        case .loading: return "Downloading..."
        case .success: return "Downloaded"
        case .error: return "Try Again"
        }
    }
    
    var buttonColor: Color {
        switch buttonState {
        case .idle: return .blue
        case .loading: return .blue
        case .success: return .green
        case .error: return .red
        }
    }
    
    func performOperation() {
        guard buttonState != .loading else { return }
        
        buttonState = .loading
        progress = 0
        
        // Simulate progress
        Timer.scheduledTimer(withTimeInterval: 0.1, repeats: true) { timer in
            progress += 0.05
            
            if progress >= 1.0 {
                timer.invalidate()
                
                // Simulate success/failure
                buttonState = Bool.random() ? .success : .error
                
                // Reset after delay
                DispatchQueue.main.asyncAfter(deadline: .now() + 2) {
                    buttonState = .idle
                    progress = 0
                }
            }
        }
    }
}

Advanced Animated Button with Particles

struct ParticleButtonView: View {
    @State private var isPressed = false
    @State private var particles: [Particle] = []
    @State private var animationAmount = 1.0
    
    var body: some View {
        Button {
            createParticles()
            performAction()
        } label: {
            ZStack {
                RoundedRectangle(cornerRadius: 20)
                    .fill(
                        LinearGradient(
                            colors: [.pink, .purple, .blue],
                            startPoint: .topLeading,
                            endPoint: .bottomTrailing
                        )
                    )
                    .frame(width: 200, height: 60)
                
                ForEach(particles, id: \.id) { particle in
                    Circle()
                        .fill(particle.color)
                        .frame(width: particle.size, height: particle.size)
                        .offset(particle.offset)
                        .opacity(particle.opacity)
                        .scaleEffect(particle.scale)
                }
                
                Text("Magic Button ‚ú®")
                    .foregroundColor(.white)
                    .fontWeight(.bold)
                    .scaleEffect(animationAmount)
            }
        }
        .scaleEffect(isPressed ? 0.95 : 1.0)
        .onLongPressGesture(minimumDuration: 0) { } onPressingChanged: { pressing in
            withAnimation(.easeInOut(duration: 0.1)) {
                isPressed = pressing
            }
        }
        .animation(.easeInOut(duration: 2), value: animationAmount)
        .onAppear {
            Timer.scheduledTimer(withTimeInterval: 0.1, repeats: true) { _ in
                updateParticles()
            }
        }
    }
    
    func createParticles() {
        for _ in 0..20 {
            let particle = Particle(
                id: UUID(),
                color: [.yellow, .orange, .pink, .purple].randomElement()!,
                size: Double.random(in: 4...12),
                offset: CGSize(
                    width: Double.random(in: -100...100),
                    height: Double.random(in: -100...100)
                ),
                opacity: 1.0,
                scale: 1.0
            )
            particles.append(particle)
        }
        
        withAnimation(.easeOut(duration: 1.5)) {
            animationAmount = 1.2
        }
        
        DispatchQueue.main.asyncAfter(deadline: .now() + 1.5) {
            particles.removeAll()
            animationAmount = 1.0
        }
    }
    
    func updateParticles() {
        for i in particles.indices {
            particles[i].opacity -= 0.02
            particles[i].scale += 0.02
            particles[i].offset.width += Double.random(in: -2...2)
            particles[i].offset.height += Double.random(in: -2...2)
        }
        particles.removeAll { $0.opacity <= 0 }
    }
    
    func performAction() {
        // Your action here
        print("Magic happened! ‚ú®")
    }
}

struct Particle {
    let id: UUID
    let color: Color
    let size: Double
    var offset: CGSize
    var opacity: Double
    var scale: Double
}

struct GestureButtonView: View {
    @State private var scale: CGFloat = 1.0
    @State private var rotation: Double = 0
    @State private var longPressTriggered = false
    
    var body: some View {
        Button("Gesture Button") {
            // Regular tap action
            quickAction()
        }
        .scaleEffect(scale)
        .rotationEffect(.degrees(rotation))
        .background(longPressTriggered ? .green : .blue)
        .foregroundColor(.white)
        .padding()
        .clipShape(RoundedRectangle(cornerRadius: 12))
        .simultaneousGesture(
            // Long press gesture
            LongPressGesture(minimumDuration: 1.0)
                .onChanged { _ in
                    withAnimation(.easeInOut(duration: 0.3)) {
                        scale = 1.2
                        longPressTriggered = true
                    }
                }
                .onEnded { _ in
                    longPressAction()
                    withAnimation(.spring()) {
                        scale = 1.0
                        rotation += 360
                    }
                    
                    DispatchQueue.main.asyncAfter(deadline: .now() + 1) {
                        longPressTriggered = false
                    }
                }
        )
        .onTapGesture {
            withAnimation(.easeInOut(duration: 0.1)) {
                scale = 0.9
            }
            withAnimation(.easeInOut(duration: 0.1).delay(0.1)) {
                scale = 1.0
            }
        }
    }
    
    func quickAction() {
        print("Quick tap!")
    }
    
    func longPressAction() {
        print("Long press completed!")
    }
}

struct AccessibleButtonView: View {
    @Environment(\.dynamicTypeSize) var dynamicTypeSize
    @Environment(\.colorScheme) var colorScheme
    
    let title: String
    let action: () -> Void
    
    var body: some View {
        Button(action: action) {
            Text(title)
                .font(.body)
                .fontWeight(.semibold)
                .padding(.horizontal, horizontalPadding)
                .padding(.vertical, verticalPadding)
                .frame(minHeight: minimumTapTarget)
                .background(buttonBackgroundColor)
                .foregroundColor(buttonForegroundColor)
                .clipShape(RoundedRectangle(cornerRadius: 12))
                .overlay(
                    RoundedRectangle(cornerRadius: 12)
                        .stroke(borderColor, lineWidth: 2)
                )
        }
        .accessibilityLabel(title)
        .accessibilityHint("Double tap to activate")
        .accessibilityRole(.button)
        .accessibilityAddTraits(.isButton)
    }
    
    // Adaptive sizing based on Dynamic Type
    var horizontalPadding: CGFloat {
        dynamicTypeSize >= .accessibility1 ? 32 : 24
    }
    
    var verticalPadding: CGFloat {
        dynamicTypeSize >= .accessibility1 ? 20 : 16
    }
    
    // Minimum tap target of 44x44 points (iOS HIG)
    var minimumTapTarget: CGFloat {
        44
    }
    
    // High contrast colors
    var buttonBackgroundColor: Color {
        colorScheme == .dark ? .white : .black
    }
    
    var buttonForegroundColor: Color {
        colorScheme == .dark ? .black : .white
    }
    
    var borderColor: Color {
        .clear
    }
}

Performance-Optimized Button with Complex Animation

struct PerformantAnimatedButton: View {
    @State private var isAnimating = false
    @State private var rippleOffset: CGFloat = 0
    @State private var rippleOpacity: Double = 0
    
    let title: String
    let action: () -> Void
    
    var body: some View {
        Button(action: {
            triggerRippleEffect()
            action()
        }) {
            ZStack {
                // Base button
                RoundedRectangle(cornerRadius: 16)
                    .fill(.blue)
                    .frame(width: 200, height: 60)
                
                // Ripple effect layer
                Circle()
                    .fill(.white)
                    .frame(width: rippleOffset, height: rippleOffset)
                    .opacity(rippleOpacity)
                    .blendMode(.overlay)
                    .animation(.easeOut(duration: 0.6), value: rippleOffset)
                    .animation(.easeOut(duration: 0.3), value: rippleOpacity)
                
                // Button text
                Text(title)
                    .foregroundColor(.white)
                    .fontWeight(.semibold)
                    .scaleEffect(isAnimating ? 1.05 : 1.0)
                    .animation(.spring(response: 0.3, dampingFraction: 0.7), value: isAnimating)
            }
        }
        .buttonStyle(PlainButtonStyle()) // Removes default button styling
        .drawingGroup() // Optimize complex views
    }
    
    func triggerRippleEffect() {
        // Reset animation state
        rippleOffset = 0
        rippleOpacity = 0.3
        
        withAnimation(.easeOut(duration: 0.6)) {
            rippleOffset = 300
        }
        
        withAnimation(.easeOut(duration: 0.3)) {
            rippleOpacity = 0
        }
        
        // Text animation
        withAnimation(.spring(response: 0.3, dampingFraction: 0.7)) {
            isAnimating = true
        }
        
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.1) {
            withAnimation(.spring(response: 0.3, dampingFraction: 0.7)) {
                isAnimating = false
            }
        }
    }
}

struct LoadingButtonView: View {

        Button {
            startLoading()
        } label: {
            HStack {

                    ProgressView()
                        .scaleEffect(0.8)
                        .progressViewStyle(CircularProgressViewStyle(tint: .white))
                }
                Text(isLoading ? "Loading..." : "Load Data")

        }
        .buttonStyle(.borderedProminent)

            isLoading = false
        }
    }
}
