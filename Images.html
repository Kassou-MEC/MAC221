HARD

Complete AsyncImage with All States

AsyncImage(url: URL(string: imageURL)) { phase in
    switch phase {
    case .empty:
        // Loading state
        VStack {
            ProgressView()
                .scaleEffect(1.5)
            Text("Loading...")
                .font(.caption)
                .foregroundColor(.secondary)
                .padding(.top, 8)
        }
        .frame(width: 200, height: 200)
        .background(.gray.opacity(0.1))
        .cornerRadius(12)
        
    case .success(let image):
        // Successfully loaded image
        image
            .resizable()
            .aspectRatio(contentMode: .fill)
            .frame(width: 200, height: 200)
            .clipShape(RoundedRectangle(cornerRadius: 12))
            .transition(.opacity.combined(with: .scale))
            
    case .failure(_):
        // Error state
        VStack {
            Image(systemName: "photo")
                .font(.system(size: 40))
                .foregroundColor(.gray)
            Text("Failed to load")
                .font(.caption)
                .foregroundColor(.secondary)
        }
        .frame(width: 200, height: 200)
        .background(.gray.opacity(0.1))
        .cornerRadius(12)
        
    @unknown default:
        EmptyView()
    }
}
.animation(.easeInOut, value: phase)

Advanced AsyncImage with Retry Logic

struct RetryableAsyncImage: View {
    let url: URL?
    @State private var retryCount = 0
    @State private var imageKey = UUID()
    
    var body: some View {
        AsyncImage(url: url) { phase in
            switch phase {
            case .empty:
                VStack {
                    ProgressView()
                    Text("Loading image...")
                        .font(.caption)
                }
                .frame(width: 200, height: 200)
                
            case .success(let image):
                image
                    .resizable()
                    .aspectRatio(contentMode: .fit)
                    .transition(.opacity)
                    
            case .failure(_):
                VStack(spacing: 12) {
                    Image(systemName: "exclamationmark.triangle")
                        .font(.system(size: 30))
                        .foregroundColor(.orange)
                    
                    Text("Failed to load image")
                        .font(.caption)
                    
                    if retryCount < 3 {
                        Text("Tap to retry (\(retryCount)/3)")
                            .font(.caption2)
                            .foregroundColor(.blue)
                            .onTapGesture {
                                retryImage()
                            }
                    }
                }
                .frame(width: 200, height: 200)
                
            @unknown default:
                EmptyView()
            }
        }
        .id(imageKey) // Force reload when key changes
    }
    
    private func retryImage() {
        retryCount += 1
        imageKey = UUID() // Trigger reload
    }
}
Loading State When It Happens Best Practice .empty Initial loading Show progress indicator .success(image) Image loaded successfully Display image with transitions .failure(error) Network or decode error Show fallback with retry option

Interactive Image Effects

struct InteractiveImageView: View {
    @State private var isHovered = false
    @State private var rotation: Double = 0
    
    var body: some View {
        Image("artwork")
            .resizable()
            .aspectRatio(contentMode: .fill)
            .frame(width: 200, height: 200)
            .clipShape(
                RoundedRectangle(cornerRadius: isHovered ? 30 : 15)
            )
            .scaleEffect(isHovered ? 1.1 : 1.0)
            .rotationEffect(.degrees(rotation))
            .shadow(
                color: .black.opacity(isHovered ? 0.3 : 0.1),
                radius: isHovered ? 20 : 5,
                y: isHovered ? 10 : 2
            )
            .animation(.spring(response: 0.6, dampingFraction: 0.8), value: isHovered)
            .onHover { hovering in
                isHovered = hovering
            }
            .onTapGesture {
                withAnimation(.easeInOut(duration: 0.5)) {
                    rotation += 90
                }
            }
    }
}

üí° Modifier Order Matters!

Apply modifiers in the right order:

  1. .resizable() first
  2. Size and aspect ratio
  3. Clipping and shapes
  4. Overlays and borders
  5. Shadows and effects

‚ö° Performance Tips:

  • Use .clipped() after .fill to prevent overflow
  • Prefer .scaledToFit/Fill over manual sizing
  • Cache complex styled images when possible
  • Use .drawingGroup() for heavy effects

Advanced Image Gallery with Parallax

struct ParallaxImageGallery: View {
    @State private var scrollOffset: CGFloat = 0
    let images = ["image1", "image2", "image3"]
    
    var body: some View {
        ScrollView(.horizontal, showsIndicators: false) {
            HStack(spacing: 20) {
                ForEach(images.indices, id: \.self) { index in
                    GeometryReader { geometry in
                        let offset = geometry.frame(in: .global).minX
                        let parallaxOffset = offset * 0.5
                        
                        AsyncImage(url: URL(string: "https://picsum.photos/300/400?random=\(index)")) { image in
                            image
                                .resizable()
                                .aspectRatio(contentMode: .fill)
                                .offset(x: parallaxOffset)
                        } placeholder: {
                            RoundedRectangle(cornerRadius: 20)
                                .fill(.gray.opacity(0.3))
                                .overlay(
                                    ProgressView()
                                )
                        }
                        .frame(width: 250, height: 350)
                        .clipShape(RoundedRectangle(cornerRadius: 20))
                        .shadow(
                            color: .black.opacity(0.2),
                            radius: 10,
                            x: 0,
                            y: 5
                        )
                        .scaleEffect(
                            1.0 - abs(offset) / 1000
                        )
                        .opacity(
                            1.0 - abs(offset) / 800
                        )
                    }
                    .frame(width: 250)
                }
            }
            .padding(.horizontal)
        }
    }
}
