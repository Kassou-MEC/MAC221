What is MVVM?

MVVM (Model-View-ViewModel) is a software architectural pattern that facilitates separation of development of the graphical user interface from the development of the business logic or back-end logic (the data model).

Model

Represents data and business logic

→
ViewModel

Exposes data and commands for the View

→
View

Displays information to the user

Why Use MVVM?

  • Separation of Concerns: Each component has a distinct responsibility
  • Testability: ViewModels can be tested without the UI
  • Maintainability: Changes to one component don't heavily impact others
  • Reusability: ViewModels can be used with different Views

MVVM in SwiftUI

SwiftUI's declarative syntax and data binding capabilities make it particularly well-suited for the MVVM pattern. The @State, @ObservedObject, and @Published property wrappers facilitate the data flow between components.

MVVM Components

Model

The Model represents the data and business logic of your application. It's responsible for data retrieval, storage, and validation.

// Example Model in Swift
struct User: Codable {
  let id: UUID
  var name: String
  var email: String
}

ViewModel

The ViewModel acts as a bridge between the Model and the View. It transforms model data into values that can be displayed and handles user interactions.

// Example ViewModel in SwiftUI
class UserViewModel: ObservableObject {
  @Published var user: User
  @Published var isLoading = false

  func loadUser() {
    // Fetch user data from network or database
  }
}

View

The View is responsible for displaying the user interface. In SwiftUI, Views are structs that conform to the View protocol and declare what should appear on screen.

// Example View in SwiftUI
struct UserView: View {
  @StateObject private var viewModel = UserViewModel()

  var body: some View {
    VStack {
      Text(viewModel.user.name)
      Text(viewModel.user.email)
    }
    .onAppear {
      viewModel.loadUser()
    }
  }
}

Simple MVVM Example: Counter App

Let's build a simple counter application using MVVM in SwiftUI. This example demonstrates the basic data flow between components.

Model
// The Model is simple in this case - just an integer
// In more complex apps, the Model would be a struct or class
// representing your data
ViewModel
class CounterViewModel: ObservableObject {
  @Published var count = 0

  func increment() {
    count += 1
  }

  func decrement() {
    count -= 1
  }

  func reset() {
    count = 0
  }
}
View
struct CounterView: View {
  @StateObject private var viewModel = CounterViewModel()

  var body: some View {
    VStack(spacing: 20) {
      Text("Count: \(viewModel.count)")
        .font(.largeTitle)
        .foregroundColor(viewModel.count < 0 ? .red : .primary)

      HStack(spacing: 20) {
        Button("Decrement") {
          viewModel.decrement()
        }
        .buttonStyle(.bordered)

        Button("Reset") {
          viewModel.reset()
        }
        .buttonStyle(.borderedProminent)

        Button("Increment") {
          viewModel.increment()
        }
        .buttonStyle(.bordered)
      }
    }
    .padding()
  }
}

Advanced MVVM Example: User Management

This example demonstrates a more realistic MVVM implementation with networking, state management, and user interactions.

Model
struct User: Codable, Identifiable {
  let id: Int
  var name: String
  var email: String
  var isActive: Bool
}

struct UserResponse: Codable {
  let users: [User]
}
ViewModel
class UserListViewModel: ObservableObject {
  @Published var users: [User] = []
  @Published var isLoading = false
  @Published var errorMessage: String?

  private let service: UserService

  init(service: UserService = .shared) {
    self.service = service
  }

  @MainActor
  func loadUsers() async {
    isLoading = true
    errorMessage = nil

    do {
      let fetchedUsers = try await service.fetchUsers()
      self.users = fetchedUsers
    } catch {
      errorMessage = "Failed to load users: \(error.localizedDescription)"
    }

    isLoading = false
  }

  func toggleUserActive(_ user: User) {
    if let index = users.firstIndex(where: { $0.id == user.id }) {
      users[index].isActive.toggle()
    }
  }

  func deleteUser(at offsets: IndexSet) {
    users.remove(atOffsets: offsets)
  }
}
View
struct UserListView: View {
  @StateObject private var viewModel = UserListViewModel()

  var body: some View {
    NavigationView {
      List {
        if viewModel.isLoading {
          ProgressView("Loading users...")
        }

        if let error = viewModel.errorMessage {
          Text(error)
            .foregroundColor(.red)
        }

        ForEach(viewModel.users) { user in
          UserRow(user: user)
            .onTapGesture {
              viewModel.toggleUserActive(user)
            }
        }
        .onDelete(perform: viewModel.deleteUser)
      }
      .navigationTitle("Users")
      .toolbar {
        ToolbarItem(placement: .navigationBarTrailing) {
          Button("Reload") {
            Task {
              await viewModel.loadUsers()
            }
          }
        }
      }
    }
    .task {
      await viewModel.loadUsers()
    }
  }
}

struct UserRow: View {
  let user: User

  var body: some View {
    HStack {
      VStack(alignment: .leading) {
        Text(user.name)
          .font(.headline)
        Text(user.email)
          .font(.subheadline)
          .foregroundColor(.gray)
      }
      Spacer()
      Image(systemName: user.isActive ? "checkmark.circle.fill" : "circle")
        .foregroundColor(user.isActive ? .green : .gray)
    }
  }
}
